227  static void append_argument(const std::string &arg, std::string &commandline) {
228  if (!arg.empty() && arg.find_first_of(" \t\n\v\"") == arg.npos) {
229  commandline.append(arg);
230  } else {
231  // quote
232  commandline.push_back('"');
233  for (auto it = arg.begin(); it != arg.end(); ++it) {
234  unsigned num_slashes = 0;
235 
236  while (it != arg.end() && *it == '\\') {
237  ++it;
238  ++num_slashes;
239  }
240 
241  if (it == arg.end()) {
242  //
243  // Escape all backslashes, but let the terminating
244  // double quotation mark we add below be interpreted
245  // as a metacharacter.
246  //
247  commandline.append(num_slashes * 2, '\\');
248  break;
249  } else if (*it == '"') {
250  //
251  // Escape all backslashes and the following
252  // double quotation mark.
253  //
254  commandline.append(num_slashes * 2 + 1, '\\');
255  commandline.push_back(*it);
256  } else {
257  //
258  // Backslashes aren't special here.
259  //

261  commandline.push_back(*it);

208  void notify(bool broadcast) {
209  int signals;
210  {
211  // lock prcondition_baseing new waiters from being signalled
212  std::lock_guard<decltype(unblock_lock_)> unblocklock(unblock_lock_);
213 
214  if ( waiters_->unblock != 0 ) {
215  if ( waiters_->blocked == 0) {
216  // nothing to signal
217  return;
218  }
219  if (broadcast) {
220  // move blocked to signals/unblock
221  signals = waiters_->blocked;
222  waiters_->unblock += signals;
223  waiters_->blocked = 0;
224  } else {
225  // only signal one
226  signals = 1;
227  waiters_->unblock++;
228  waiters_->blocked--;
229  }
230  } else if ( waiters_->blocked > waiters_->gone ) {
231  block_lock_.wait(); // close the gate
232  if ( waiters_->gone != 0) {
233  // waiters gone are no longer blocked
234  waiters_->blocked -= waiters_->gone;
235  waiters_->gone = 0;
236  }
237  if (broadcast) {
238  // move blocked to unblock/signals
239  signals = waiters_->blocked;
240  waiters_->unblock = signals;
241  waiters_->blocked = 0;
242  } else {
243  signals = 1;
244  waiters_->unblock = 1;
245  waiters_->blocked--;
246  }
247  } else {
248  return;
249  }
250  }
251  // post a number of signals
252  while (signals > 0) {
253  --signals;
254  block_queue_.notify();
255  }

269  bool timeout, const std::chrono::time_point<Clock, Duration>& abs_time) {

277  ++(waiters_->blocked);
278  }

281  // unlock lock, and protect so will be relocked out of scope

283  std::lock_guard<lock_inverter<std::unique_lock<cpen333::process::mutex>>> external_unlock(ilock);

287  if (timeout) {

289  } else {
290  block_queue_.wait();
291  }
292 
293  // reduce waiters count and check if we were the last in a notify_all
294  {
295  std::lock_guard<cpen333::process::mutex> unblocklock(unblock_lock_);
296  signals_left = waiters_->unblock;
297 
298  if (signals_left != 0) {
299  if (timed_out) {
300  if (waiters_->blocked != 0) {
301  --(waiters_->blocked);
302  } else {

304  }
305  }
306

308  if (waiters_->blocked != 0) {
309  block_lock_.notify(); // allow one more through
310  signals_left = 0; // let him take care of rest
311  } else {

313  waiters_gone = waiters_->gone;
314  if (waiters_gone != 0) {

316  }
317  } // blocked
318  } // last to unblock
319  } // signals left
320

325  waiters_->blocked -= waiters_->gone;
326  waiters_->gone = 0;
327  }
328  } // unblock lock is unlocked

339  }
